  if (length(missing) > 0) {
    message("Installing missing packages: ", paste(missing, collapse = ", "))
    install.packages(missing, repos = "https://cloud.r-project.org", quiet = TRUE)
  }
}

# Parse year from ig_date ("YYYY-MM-DD" character string)
# Assign dominant severity class based on which percentage is highest
mtbs_parsed <- mtbs_raw |>
  mutate(
    year = as.integer(substr(ig_date, 1, 4)),

    # Dominant severity: highest of low/mod/high percentage wins
    # (incgreen_t and nodata_t are not traditional severity classes)
    dominant_severity = case_when(
      high_t >= mod_t & high_t >= low_t & high_t > 0 ~ "High",
      mod_t  >= high_t & mod_t  >= low_t & mod_t  > 0 ~ "Moderate",
      low_t  >= high_t & low_t  >= mod_t & low_t  > 0 ~ "Low",
      TRUE ~ "Other"  # fires where no severity class dominates (e.g. increased greenness)
    ),
    dominant_severity = factor(dominant_severity,
                               levels = c("Low", "Moderate", "High", "Other"))
  )

cat("Year range in raw data:", min(mtbs_parsed$year, na.rm = TRUE),
    "–", max(mtbs_parsed$year, na.rm = TRUE), "\n")

Code

Plot 2 — Fires Per Year (Temporal Trend)

Annual wildfire count across the Western US, 2000–2023, with a linear trend line. This reveals whether fire frequency is increasing over time — a key consideration for the staggered DiD design (more treatment cohorts = better identification).

Code 
fires_per_year <- mtbs_west |>
  st_drop_geometry() |>
  count(year, name = "n_fires")

# Sanity check: should have one row per year across the full range
expected_years <- YEAR_MIN:YEAR_MAX
missing_years  <- setdiff(expected_years, fires_per_year$year)
if (length(missing_years) > 0) {
  message("Years with zero fires (will add as 0): ", paste(missing_years, collapse = ", "))
  fires_per_year <- fires_per_year |>
    complete(year = expected_years, fill = list(n_fires = 0))
}

p2 <- ggplot(fires_per_year, aes(x = year, y = n_fires)) +
  geom_col(fill = "#fc8d59", color = "white", linewidth = 0.3, alpha = 0.85) +
  # Linear trend line to visualize direction of change
  geom_smooth(method = "lm", se = TRUE,
              color = "#d73027", fill = "#d73027", alpha = 0.15,
              linewidth = 1) +
  scale_x_continuous(
    breaks = seq(YEAR_MIN, YEAR_MAX, by = 2),
    expand = c(0.01, 0)
  ) +
  scale_y_continuous(
    labels = comma,
    expand = c(0, 0),
    limits = c(0, max(fires_per_year$n_fires) * 1.15)
  ) +
  labs(
    title   = "Annual Wildfire Count — Western United States",
    x       = "Year",
    y       = "Number of wildfires",
    caption = "Source: USGS MTBS. Wildfires ≥ 1,000 acres only. Red band = 95% CI around linear trend."
  ) +
  theme_minimal(base_size = 12) +
  theme(
    plot.title      = element_text(face = "bold", size = 14),
    plot.caption    = element_text(color = "grey50", size = 9),
    panel.grid.minor = element_blank(),
    axis.title      = element_text(size = 11),
    axis.text       = element_text(size = 10)
  )

# ---- Sanity checks on dNBR threshold fields --------------------------------
# These checks guard against the most common misuse: treating threshold values
# (dNBR index units, typically 50–600) as if they were percentages (0–100).

valid_thresh <- mtbs_west |>
  st_drop_geometry() |>
  filter(thresholds_valid)

# Check 1: Threshold orientation.
# MTBS fires can be mapped on either axis of the dNBR spectrum:
#   "Burned" fires:             low_t < mod_t < high_t (thresholds increase toward high severity)
#   "Increased Greenness" fires: low_t > mod_t > high_t (thresholds decrease, negative dNBR)
# Both orderings are valid. What is NOT valid is a MIXED ordering (non-monotone),
# which would indicate corrupted threshold values.
valid_thresh <- valid_thresh |>
  mutate(
    thresh_orient = case_when(
      low_t < mod_t & mod_t < high_t ~ "burned",       # typical positive-dNBR fire
      low_t > mod_t & mod_t > high_t ~ "greenness",    # increased-greenness fire
      TRUE                            ~ "mixed"         # non-monotone — data problem
    )
  )

Plot 4 — Burn Severity: dNBR and RdNBR Distributions

# =============================================================================
# BURN SEVERITY METRICS — dNBR vs. RdNBR
#
# dNBR  (differenced Normalized Burn Ratio):
#   dNBR = (preFireNBR − postFireNBR) × 1000
#   Captures spectral change caused by fire. Positive values = vegetation
#   loss (burning). Stored in MTBS/BSP as integer index units (×1000 scale),
#   so values typically range from −500 (increased greenness) to 1300+ (high
#   severity). A useful absolute measure of spectral change per fire pixel.
#
#   LIMITATION: pre-fire biomass confounds cross-fire comparison. A dNBR of
#   300 in dense old-growth forest represents less relative canopy destruction
#   than 300 in sparse shrubland, because old-growth had more NBR to lose.
#
# RdNBR (Relativized dNBR; Miller & Thode 2007):
#   RdNBR = dNBR / sqrt(|preFireNBR / 1000|)
#   Divides by the square root of pre-fire NBR (rescaled to the 0–1 range),
#   normalizing for vegetation density before the fire. This makes severity
#   more comparable across fires in different biomes and vegetation types —
#   the preferred continuous treatment variable for a cross-fire DiD study.
#
# Data: USGS Burn Severity Program (BSP) CBI field plots — geospatially
# validated measurement points sampled within fire perimeters by field crews.
# BSP stores raw dNBR and pre-fire NBR per plot, enabling RdNBR computation.
# These are the only per-point raw dNBR observations available in this
# download without fetching individual MTBS fire severity rasters.
# =============================================================================

BSP_PATH <- here("data/raw/mtbs/burn_severity_perimeter_data/bsp_perims_DD.shp")

bsp_raw <- st_read(BSP_PATH, quiet = TRUE)
names(bsp_raw) <- tolower(names(bsp_raw))

# Keep only CBI field plots with valid dNBR and pre-fire NBR.
# Exclude sentinel 9999 (no data) and |preNBR| ≤ 10 (avoids RdNBR blow-up
# from near-zero pre-fire NBR in bare/sparse areas).
cbi_raw <- bsp_raw |>
  filter(
    map_prog   == "CBI",
    !is.na(dnbr_val),  !is.na(prenbr_val),
    dnbr_val   != 9999, prenbr_val != 9999,
    abs(prenbr_val) > 10
  ) |>
  mutate(
    # RdNBR: normalize by sqrt(|preNBR|) on the 0–1 NBR scale
    rdnbr = dnbr_val / sqrt(abs(prenbr_val / 1000))
  )

# Spatial filter to Western US (BSP is NAD83, matches western_states_sf CRS)
cbi_raw  <- st_transform(cbi_raw, st_crs(western_states_sf))

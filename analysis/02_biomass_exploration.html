.install_if_missing <- function(pkgs) {
  missing <- pkgs[!pkgs %in% rownames(installed.packages())]
  if (length(missing) > 0) {
    message("Installing: ", paste(missing, collapse = ", "))
    install.packages(missing, repos = "https://cloud.r-project.org", quiet = TRUE)
  }
}
.install_if_missing(c("tidyverse", "here", "scales", "glue", "png", "patchwork"))

2. GEE Extraction

The biomass time series is extracted using the LandTrendR spectral-temporal segmentation algorithm in Google Earth Engine. The extraction script is scripts/python/01_extract_biomass_gee.py.

Before rendering this document for the first time:

  1. Set your GEE project ID in 01_extract_biomass_gee.py (GEE_PROJECT = "...")
  2. Authenticate: run earthengine authenticate in a terminal
  3. The chunk below will run the script automatically if output/biomass_timeseries.csv does not yet exist
Code 
if (!file.exists(CSV_PATH)) {
  message("output/biomass_timeseries.csv not found — running GEE extraction.")
  message("This will take several minutes depending on sample size and GEE queue.")

  # Find Anaconda Python (has earthengine-api, geemap, geopandas)
  python_paths <- c(
    "C:/Users/shaht/anaconda3/python.exe",
    Sys.which("python3"),
    Sys.which("python")
  )
  python_exe <- python_paths[file.exists(python_paths)][1]

  if (is.na(python_exe)) {
    stop("Python not found. Set the path manually in the run-extraction chunk.")
  }

  result <- system2(python_exe, args = PYTHON_SCRIPT,
                    stdout = TRUE, stderr = TRUE)
  cat(result, sep = "\n")

  if (!file.exists(CSV_PATH)) {
    stop(
      "Extraction did not produce output/biomass_timeseries.csv.\n",
      "Check GEE authentication and GEE_PROJECT in 01_extract_biomass_gee.py."
    )
  }
  message("Extraction complete.")
} else {
  message("CSV found at ", CSV_PATH, " — skipping extraction.")
}

Plot 5 — Biomass Change Over Time

Mean annual fitted NBR across all sampled fire sites, 2000–2023. Shows the overall trend in biomass conditions across the Western US fire landscape.

Code 
# Aggregate by year: mean ± SE across all fire sites
annual_nbr <- biomass |>
  group_by(year) |>
  summarise(
    mean_nbr = mean(display_nbr),
    se_nbr   = sd(display_nbr) / sqrt(n()),
    n_fires  = n_distinct(event_id),
    .groups  = "drop"
  )

# Sanity checks on aggregate
stopifnot("Annual NBR should not be all NA" = !all(is.na(annual_nbr$mean_nbr)))
stopifnot("X-axis must span study period"   =
    all(c(START_YEAR, END_YEAR) %in% annual_nbr$year))

p5 <- ggplot(annual_nbr, aes(x = year, y = mean_nbr)) +
  # Uncertainty band (± 1 SE)
  geom_ribbon(aes(ymin = mean_nbr - se_nbr,
                  ymax = mean_nbr + se_nbr),
              fill = "#2c7bb6", alpha = 0.2) +
  # Annual mean line + points
  geom_line(color = "#2c7bb6", linewidth = 1) +
  geom_point(color = "#2c7bb6", size = 2.5) +
  # Linear trend line (dashed)
  geom_smooth(method = "lm", se = FALSE,
              color = "black", linetype = "dashed", linewidth = 0.8) +
  scale_x_continuous(
    breaks = seq(START_YEAR, END_YEAR, by = 2),
    expand = c(0.01, 0)
  ) +
  scale_y_continuous(labels = number_format(accuracy = 0.001)) +
  labs(
    title    = "Annual Mean Fitted NBR — Western US Wildfire Sites (2000–2023)",
    subtitle = glue("LandTrendR fitted values across {n_fires} sampled MTBS wildfire perimeters"),
    x        = "Year",
    y        = "Mean fitted NBR",
    caption  = paste(
      "NBR displayed in natural scale (higher = more vegetation).",
      "Shaded band = ± 1 SE. Dashed line = linear trend.",
      "Source: USGS MTBS + Landsat via Google Earth Engine."
    )
  ) +
  theme_minimal(base_size = 12) +
  theme(
    plot.title    = element_text(face = "bold", size = 14),
    plot.subtitle = element_text(color = "grey40", size = 11),
    plot.caption  = element_text(color = "grey50", size = 9),
    panel.grid.minor = element_blank(),
    axis.title    = element_text(size = 11),
    axis.text     = element_text(size = 10)
  )

print(p5)

Plot 6 — Biomass Before and After Fire Events

Event-study plot showing mean fitted NBR relative to the fire year (year 0). Each fire contributes one time series centered on its ignition year. This is the EDA preview of the core identification pattern the Callaway-Sant’Anna estimator will formally test.

Code 
# Compute relative year and filter to event window
biomass_event <- biomass |>
  mutate(rel_year = year - fire_year) |>
  filter(rel_year >= -5, rel_year <= 10)

# How many fires contribute data at each relative year?
coverage <- biomass_event |>
  group_by(rel_year) |>
